import java.net.*; 
import java.io.*; 
import java.util.*; 
public class ChatHubServer { 
 static List<Client> clients = Collections.synchronizedList(new ArrayList<>()); 
 public static void main(String[] a) throws Exception { 
 try (ServerSocket s = new ServerSocket(5052)) { 
 System.out.println("ChatHubServer running on 5052"); 
 while (true) { 
 Socket c = s.accept(); 
 Client cl = new Client(c); 
 clients.add(cl); 
 new Thread(cl).start(); 
 }
 } 
 } 
 static void broadcast(String msg, Client from) { 
 synchronized (clients) { 
 for (Client c : clients) 
 if (c != from) c.send(msg); 
 } 
 } 
 static void remove(Client c) { 
 clients.remove(c); 
 } 
 static class Client implements Runnable { 
 Socket sock; BufferedReader r; PrintWriter w; String nick = "Guest"; 
 Client(Socket s) throws IOException { 
 sock = s; 
 r = new BufferedReader(new InputStreamReader(s.getInputStream()));  w = new PrintWriter(s.getOutputStream(), true);  } 
 public void run() { 
 try { 
 w.println("Enter your nickname:"); 
 nick = r.readLine(); 
 broadcast(nick + " joined the chat.", this);  String line; 
 while ((line = r.readLine()) != null) { 
 if ("exit".equalsIgnoreCase(line.trim())) break;  broadcast(nick + ": " + line, this); 
 } 
 } catch (IOException e) { 
 } finally { 
 try { sock.close(); } catch (IOException e) {}  remove(this); 
 broadcast(nick + " left the chat.", this); 
 } 
 } 
 void send(String msg) { w.println(msg); } 
 } 
}

import java.net.*; 
import java.io.*; 
import java.util.Random; 
public class NumberSumServer { 
 public static void main(String[] a) throws Exception { 
 try (ServerSocket s = new ServerSocket(5053)) { 
 System.out.println("NumberSumServer listening on 5053"); 
 while (true) { 
 Socket c = s.accept(); 
 new Thread(() -> handle(c)).start(); 
 } 
 } 
 } 
 static void handle(Socket c) { 
 try (DataInputStream dis = new DataInputStream(c.getInputStream());  DataOutputStream dos = new DataOutputStream(c.getOutputStream())) {  int n = dis.readInt(); 
 if (n <= 0) n = 5; 
 Random r = new Random(); 
 long sum = 0; 
 dos.writeInt(n); 
 for (int i = 0; i < n; i++) { 
 int v = r.nextInt(100); 
 dos.writeInt(v); 
 sum += v; 
 } 
 dos.writeLong(sum);
 dos.flush(); 
 } catch (IOException e) { 
 } finally { 
 try { c.close(); } catch (IOException e) {} 
 } 
 } 
}

public class FileReceiverServer { 
 public static void main(String[] a) throws Exception { 
 try (ServerSocket s = new ServerSocket(5054)) { 
 System.out.println("FileReceiverServer running on 5054");  while (true) { 
 Socket c = s.accept();
 new Thread(() -> recv(c)).start(); 
 } 
 } 
 } 
 static void recv(Socket c) { 
 try (DataInputStream dis = new DataInputStream(new BufferedInputStream(c.getInputStream())); 
 DataOutputStream dos = new DataOutputStream(c.getOutputStream())) {  String name = dis.readUTF(); 
 long size = dis.readLong(); 
 String out = "received_" + name; 
 try (FileOutputStream fos = new FileOutputStream(out)) { 
 byte[] b = new byte[4096]; 
 long rem = size; 
 int r; 
 while (rem > 0 && (r = dis.read(b, 0, (int) Math.min(b.length, rem))) != -1) {  fos.write(b, 0, r); 
 rem -= r; 
 } 
 } 
 dos.writeUTF("OK"); 
 dos.flush(); 
 } catch (IOException e) { 
 System.err.println("Transfer error: " + e.getMessage()); 
 } finally { 
 try { c.close(); } catch (IOException ignored) {} 
 } 
 } 
}

public class UDPChatServer { 
 public static void main(String[] a) throws Exception { 
 DatagramSocket s = new DatagramSocket(9999); 
 Set<InetSocketAddress> clients = ConcurrentHashMap.newKeySet(); 
 byte[] buf = new byte[2048]; 
 System.out.println("UDPChatServer running on 9999"); 
 while (true) { 
 DatagramPacket p = new DatagramPacket(buf, buf.length); 
 s.receive(p); 
 String msg = new String(p.getData(), 0, p.getLength()); 
 InetSocketAddress addr = new InetSocketAddress(p.getAddress(), p.getPort());  clients.add(addr); 
 String out = "[" + p.getAddress().getHostAddress() + ":" + p.getPort() + "] " + msg;  byte[] data = out.getBytes(); 
 for (InetSocketAddress c : clients) 
 s.send(new DatagramPacket(data, data.length, c)); 
 } 
 } 
}
